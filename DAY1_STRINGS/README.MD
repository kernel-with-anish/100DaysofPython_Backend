def longest_unique(s):
    seen, l, res = set(), 0, 0
    for r in range(len(s)):
        while s[r] in seen:
            seen.remove(s[l]); l += 1
        seen.add(s[r])
        res = max(res, r - l + 1)
    return res

def valid_parentheses(s):
    stack, mapping = [], {')':'(', ']':'[', '}':'{'}
    for c in s:
        if c in mapping:
            if not stack or stack.pop() != mapping[c]: return False
        else:
            stack.append(c)
    return not stack
